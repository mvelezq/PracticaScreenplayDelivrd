function camelcase(inputstring) {
    var a = inputstring.split('_'), i;
    s = [];
    for (i = 0; i &lt; a.length; i++) {
        s.push(a[i].charAt(0).toUpperCase() + a[i].substring(1));
    }
    s = s.join('');
    return s;
}
